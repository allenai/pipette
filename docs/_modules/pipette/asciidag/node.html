[docs]class Node(object):
    def __init__(self, item, parents=None):
        self.item = item
        if parents is None:
            parents = []
        try:
            iter(parents)
        except TypeError:
            raise InvalidNodeException(str(item))
        self.parents = parents

    def __str__(self):
        return str(self.item)
